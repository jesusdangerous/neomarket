import requests
from typing import Optional, List, Dict, Any
from datetime import datetime
from urllib.parse import urljoin

from models import (
    FavoritesResponse, FavoriteMutationResponse, SubscribeRequest,
    SubscriptionResponse, CartResponse, AddCartItemRequest,
    UpdateCartItemRequest, CartMutationResponse, CartValidationResponse,
    BannerListResponse, BannerEventsRequest, CollectionsResponse,
    CollectionProductsResponse, AlsoBoughtResponse, SimilarProductsResponse,
    ErrorResponse, NotificationType, BannerEventType

class NeoMarketAPI:
    """
    Клиент для NeoMarket B2C API (Корзина + Избранное + Главная)
    
    Документация API: OpenAPI спецификация от Interface Team
    Версия: 0.1
    """
    
    def __init__(self, base_url: str, token: Optional[str] = None):
        """
        Инициализация клиента API
        
        Args:
            base_url: Базовый URL API (например, https://api.neomarket.ru)
            token: JWT токен для авторизации (опционально)
        """
        self.base_url = base_url.rstrip('/')
        self.session = requests.Session()
        
        if token:
            self.session.headers.update({
                'Authorization': f'Bearer {token}'
            })
    
    def _make_url(self, path: str) -> str:
        """Формирует полный URL для запроса"""
        return urljoin(f"{self.base_url}/", path.lstrip('/'))
    
    def _handle_response(self, response: requests.Response, model_class=None):
        """
        Обрабатывает ответ от API
        
        Args:
            response: Объект ответа requests
            model_class: Класс модели для парсинга ответа (опционально)
        
        Returns:
            Распарсенный ответ или None для 204 статуса
        """
        if response.status_code == 204:
            return None
        
        if not response.ok:
            error_data = response.json()
            raise APIError(
                code=error_data.get('code', 'UNKNOWN_ERROR'),
                message=error_data.get('message', 'Unknown error'),
                details=error_data.get('details'),
                status_code=response.status_code
            )
        
        if model_class and response.content:
            data = response.json()
            return self._parse_response(data, model_class)
        
        return response.json() if response.content else None
    
    def _parse_response(self, data: Dict[str, Any], model_class):
        """Парсит JSON ответ в объект модели"""
        # Здесь можно добавить более сложную логику парсинга
        # Для простоты возвращаем как есть, в реальном проекте используйте dataclass-json
        return data
    
    # ========== ИЗБРАННОЕ ==========
    
    def get_favorites(self, limit: int = 20, offset: int = 0) -> FavoritesResponse:
        """
        Получение списка избранных товаров
        
        Args:
            limit: Максимальное количество элементов (1-100)
            offset: Смещение для пагинации
        
        Returns:
            FavoritesResponse: Список избранных товаров
        
        Raises:
            APIError: При ошибках API (400, 401, 503)
        """
        response = self.session.get(
            self._make_url('/api/v1/favorites'),
            params={'limit': limit, 'offset': offset}
        )
        return self._handle_response(response, FavoritesResponse)
    
    def add_to_favorites(self, product_id: str, user_id: str) -> FavoriteMutationResponse:
        """
        Добавление товара в избранное
        
        Args:
            product_id: UUID товара
            user_id: UUID пользователя
        
        Returns:
            FavoriteMutationResponse: Результат добавления
        
        Raises:
            APIError: При ошибках API (400, 401, 404, 503)
        """
        response = self.session.post(
            self._make_url(f'/api/v1/favorites/{product_id}'),
            params={'user_id': user_id}
        )
        return self._handle_response(response, FavoriteMutationResponse)
    
    def remove_from_favorites(self, product_id: str, user_id: str) -> None:
        """
        Удаление товара из избранного
        
        Args:
            product_id: UUID товара
            user_id: UUID пользователя
        
        Raises:
            APIError: При ошибках API (400, 401, 404, 503)
        """
        response = self.session.delete(
            self._make_url(f'/api/v1/favorites/{product_id}'),
            params={'user_id': user_id}
        )
        self._handle_response(response)
    
    def subscribe_to_product(
        self, 
        product_id: str, 
        notify_on: List[NotificationType]
    ) -> SubscriptionResponse:
        """
        Подписка на уведомления о товаре
        
        Args:
            product_id: UUID товара
            notify_on: Список типов уведомлений
        
        Returns:
            SubscriptionResponse: Информация о подписке
        
        Raises:
            APIError: При ошибках API (400, 401, 404, 409, 500, 503)
        """
        request = SubscribeRequest(notify_on=notify_on)
        response = self.session.post(
            self._make_url(f'/api/v1/favorites/{product_id}/subscribe'),
            json={'notify_on': [n.value for n in notify_on]}
        )
        return self._handle_response(response, SubscriptionResponse)
    
    # ========== КОРЗИНА ==========
    
    def get_cart(
        self, 
        user_id: Optional[str] = None, 
        session_id: Optional[str] = None
    ) -> CartResponse:
        """
        Просмотр корзины с актуальными ценами
        
        Args:
            user_id: UUID пользователя (для авторизованных)
            session_id: ID гостевой сессии
        
        Returns:
            CartResponse: Содержимое корзины
        
        Raises:
            APIError: При ошибках API (400, 401, 500, 503)
        """
        headers = {}
        if user_id:
            headers['X-User-Id'] = user_id
        if session_id:
            headers['X-Session-Id'] = session_id
        
        response = self.session.get(
            self._make_url('/api/v1/cart'),
            headers=headers
        )
        return self._handle_response(response, CartResponse)

    def add_cart_item(
        self,
        sku_id: str,
        quantity: int,
        user_id: Optional[str] = None,
        session_id: Optional[str] = None
    ) -> CartMutationResponse:
        """
        Добавить товар в корзину
        
        Args:
            sku_id: UUID SKU
            quantity: Количество (минимум 1)
            user_id: UUID пользователя (для авторизованных)
            session_id: ID гостевой сессии
        
        Returns:
            CartMutationResponse: Результат добавления
        
        Raises:
            APIError: При ошибках API (400, 401, 404, 410, 422, 500, 503)
        """
        headers = {}
        if user_id:
            headers['X-User-Id'] = user_id
        if session_id:
            headers['X-Session-Id'] = session_id
        
        request = AddCartItemRequest(sku_id=sku_id, quantity=quantity)
        response = self.session.post(
            self._make_url('/api/v1/cart/items'),
            headers=headers,
            json={'sku_id': sku_id, 'quantity': quantity}
        )
        return self._handle_response(response, CartMutationResponse)

    def send_banner_events(
        self,
        events: List[Dict[str, Any]]
    ) -> None:
        """
        Отправить события аналитики баннеров (показы/клики)
        
        Args:
            events: Список событий. Каждое событие содержит:
                   - banner_id: UUID баннера
                   - event: "impression" или "click"
                   - timestamp: datetime в ISO формате
        
        Raises:
            APIError: При ошибках API (400, 500)
        """
        formatted_events = []
        for event in events:
            formatted_event = {
                'banner_id': event['banner_id'],
                'event': event['event'].value if isinstance(event['event'], BannerEventType) else event['event'],
                'timestamp': event['timestamp'].isoformat() if isinstance(event['timestamp'], datetime) else event['timestamp']
            }
            formatted_events.append(formatted_event)
        
        response = self.session.post(
            self._make_url('/api/v1/banner-events'),
            json={'events': formatted_events}
        )
        self._handle_response(response)

class APIError(Exception):
    """Исключение для ошибок API"""
    
    def __init__(self, code: str, message: str, details: Optional[Dict] = None, status_code: int = 500):
        self.code = code
        self.message = message
        self.details = details
        self.status_code = status_code
        super().__init__(f"[{code}] {message}")

"""
Асинхронный клиент для NeoMarket API с использованием httpx
Требует установки: pip install httpx
"""

import httpx
from typing import Optional, List, Dict, Any
from datetime import datetime
from models import (
    FavoritesResponse, FavoriteMutationResponse, SubscribeRequest,
    SubscriptionResponse, CartResponse, AddCartItemRequest,
    UpdateCartItemRequest, CartMutationResponse, CartValidationResponse,
    BannerListResponse, CollectionsResponse, CollectionProductsResponse,
    AlsoBoughtResponse, SimilarProductsResponse, NotificationType,
    ErrorResponse, BannerEventType
)
from client import APIError


class AsyncNeoMarketAPI:
    """
    Асинхронный клиент для NeoMarket B2C API
    """
    
    def __init__(self, base_url: str, token: Optional[str] = None):
        self.base_url = base_url.rstrip('/')
        self.token = token
        self._client = None
    
    async def __aenter__(self):
        headers = {}
        if self.token:
            headers['Authorization'] = f'Bearer {self.token}'
        
        self._client = httpx.AsyncClient(
            base_url=self.base_url,
            headers=headers,
            timeout=30.0
        )
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self._client:
            await self._client.aclose()
    
    async def _request(self, method: str, path: str, **kwargs):
        """Выполняет HTTP запрос"""
        response = await self._client.request(method, path, **kwargs)
        
        if response.status_code == 204:
            return None
        
        if not response.is_success:
            try:
                error_data = response.json()
            except:
                error_data = {}
            
            raise APIError(
                code=error_data.get('code', 'UNKNOWN_ERROR'),
                message=error_data.get('message', 'Unknown error'),
                details=error_data.get('details'),
                status_code=response.status_code
            )
        
        if response.content:
            return response.json()
        return None
    
    # ===== ИЗБРАННОЕ =====
    
    async def get_favorites(self, limit: int = 20, offset: int = 0) -> Dict[str, Any]:
        """Получение списка избранных товаров"""
        return await self._request(
            'GET',
            '/api/v1/favorites',
            params={'limit': limit, 'offset': offset}
        )
    
    async def add_to_favorites(self, product_id: str, user_id: str) -> Dict[str, Any]:
        """Добавление товара в избранное"""
        return await self._request(
            'POST',
            f'/api/v1/favorites/{product_id}',
            params={'user_id': user_id}
        )
    
    async def remove_from_favorites(self, product_id: str, user_id: str) -> None:
        """Удаление товара из избранного"""
        await self._request(
            'DELETE',
            f'/api/v1/favorites/{product_id}',
            params={'user_id': user_id}
        )
    
    async def subscribe_to_product(
        self,
        product_id: str,
        notify_on: List[NotificationType]
    ) -> Dict[str, Any]:
        """Подписка на уведомления о товаре"""
        return await self._request(
            'POST',
            f'/api/v1/favorites/{product_id}/subscribe',
            json={'notify_on': [n.value for n in notify_on]}
        )
    
    # ===== КОРЗИНА =====
    
    async def get_cart(
        self,
        user_id: Optional[str] = None,
        session_id: Optional[str] = None
    ) -> Dict[str, Any]:
        """Просмотр корзины"""
        headers = {}
        if user_id:
            headers['X-User-Id'] = user_id
        if session_id:
            headers['X-Session-Id'] = session_id
        
        return await self._request('GET', '/api/v1/cart', headers=headers)
    
    async def clear_cart(
        self,
        user_id: Optional[str] = None,
        session_id: Optional[str] = None
    ) -> None:
        """Очистить корзину"""
        headers = {}
        if user_id:
            headers['X-User-Id'] = user_id
        if session_id:
            headers['X-Session-Id'] = session_id
        
        await self._request('DELETE', '/api/v1/cart', headers=headers)
    
    async def add_cart_item(
        self,
        sku_id: str,
        quantity: int,
        user_id: Optional[str] = None,
        session_id: Optional[str] = None
    ) -> Dict[str, Any]:
        """Добавить товар в корзину"""
        headers = {}
        if user_id:
            headers['X-User-Id'] = user_id
        if session_id:
            headers['X-Session-Id'] = session_id
        
        return await self._request(
            'POST',
            '/api/v1/cart/items',
            headers=headers,
            json={'sku_id': sku_id, 'quantity': quantity}
        )
    
    async def validate_cart(
        self,
        cart_item_ids: Optional[List[str]] = None
    ) -> Dict[str, Any]:
        """Валидация корзины"""
        params = {}
        if cart_item_ids:
            params['cart_item_ids'] = cart_item_ids
        
        return await self._request('GET', '/cart/validate', params=params)
    
    # ===== ГЛАВНАЯ СТРАНИЦА =====
    
    async def get_home_banners(self) -> Dict[str, Any]:
        """Получить активные баннеры"""
        return await self._request('GET', '/api/v1/home/banners')
    
    async def send_banner_events(self, events: List[Dict[str, Any]]) -> None:
        """Отправить события аналитики баннеров"""
        await self._request('POST', '/api/v1/banner-events', json={'events': events})
    
    async def get_collections(self, limit: int = 10, offset: int = 0) -> Dict[str, Any]:
        """Получить список подборок"""
        return await self._request(
            'GET',
            '/api/v1/main/collections',
            params={'limit': limit, 'offset': offset}
        )
    
    # ===== РЕКОМЕНДАЦИИ =====
    
    async def get_also_bought(self, limit: int = 10) -> Dict[str, Any]:
        """Рекомендации 'покупают вместе'"""
        return await self._request(
            'GET',
            '/api/v1/cart/also_bought',
            params={'limit': limit}
        )
    
    async def get_similar_products(self, product_id: str, limit: int = 10) -> Dict[str, Any]:
        """Похожие товары"""
        return await self._request(
            'GET',
            f'/api/v1/products/{product_id}/similar',
            params={'limit': limit}
        )
